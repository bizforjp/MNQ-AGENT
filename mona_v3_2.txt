// ============================================================================
// THE MONA v3.2 — Signal Generator (alert() transport)
// ============================================================================
// Changes from v3.0 (Option C with alertcondition + {{plot_N}} template tags):
//   - Transport layer rewritten: alertcondition() -> alert()
//   - JSON payloads built directly in Pine via str.tostring() concatenation
//   - Hidden data plots (0..41) removed — no longer needed to feed template
//   - Version bumped to "3.2" so backend can distinguish migrated signals
//   - Title differs from v3.0 so both can coexist on chart during swap
//
// Why this exists:
//   TradingView's alertcondition() template resolves only the first ~20
//   {{plot_N}} references. Plots at index >= 20 arrive at the webhook as
//   literal "{{plot_N}}" text; the backend sanitizer replaces them with 0.
//   This silently broke bar_close_ms, volume_ratio, eval fields, heartbeat
//   slots, etc. Proven 2026-04-20 via alert log payload and the alert()
//   transport-test probe (mona_alert_transport_test.txt).
//
// Zero logic changes:
//   - Same Reputation Engine (ELIGIBLE -> GROUNDED -> EXTENDED)
//   - Same entry filters (trend continuation, squeeze breakout)
//   - Same follow-through evaluation (0.5 x ATR, 4 bars)
//   - Same session window (9:30am - 3:45pm ET)
//   - Same 4-slot heartbeat machinery
//   - Same chart visuals, same tables
// ============================================================================


// ========================== INPUTS ==========================================

ema9Len   = input.int(9,   "EMA 9 Length",  group="EMAs")
ema21Len  = input.int(21,  "EMA 21 Length", group="EMAs")
ema50Len  = input.int(50,  "EMA 50 Length", group="EMAs")

rsiLen    = input.int(14,  "RSI Length",    group="StochRSI")
stochLen  = input.int(14,  "Stoch Length",  group="StochRSI")
smoothK   = input.int(3,   "Smooth K",      group="StochRSI")
smoothD   = input.int(3,   "Smooth D",      group="StochRSI")

adxLen    = input.int(14,  "ADX Length",    group="ADX")
adxMin    = input.int(20,  "ADX Minimum",   group="ADX")

atrLen    = input.int(14,   "ATR Length",              group="Risk")
slMult    = input.float(1.1, "SL Multiplier (× ATR)",  step=0.1, group="Risk")
tp1Mult   = input.float(1.5, "TP1 Multiplier (× ATR)", step=0.1, group="Risk")
tp2Mult   = input.float(2.5, "TP2 Multiplier (× ATR)", step=0.1, group="Risk")

sqzThresh   = input.float(0.5, "Squeeze Threshold (× ATR)", step=0.1, group="Squeeze")
sqzLookback = input.int(10,    "Squeeze Lookback Bars",     group="Squeeze")

srLookback = input.int(10, "S/R Pivot Lookback", group="S/R")

volFloor    = input.float(0.3, "Volume Floor (× avg)",    step=0.1, group="Filters")

followThruMult = input.float(0.5, "Follow-Through (× ATR)", step=0.1, group="Reputation")
cooldownBars   = input.int(4,     "Evaluation Window (bars)",          group="Reputation")
groundedBars   = input.int(8,     "Grounded Lockout (bars)",           group="Reputation")
extendedBars   = input.int(16,    "Extended Lockout (bars)",           group="Reputation")

showRepTable   = input.bool(true, "Show Reputation Table", group="Display")
showInfoTable  = input.bool(true, "Show Info Table",       group="Display")
showLevels     = input.bool(true, "Plot Signal Levels",    group="Display")


// ========================== CORE CALCULATIONS ===============================

ema9  = ta.ema(close, ema9Len)
ema21 = ta.ema(close, ema21Len)
ema50 = ta.ema(close, ema50Len)
vwapValue = ta.vwap(hlc3)
rsi1    = ta.rsi(close, rsiLen)
stoch_k = ta.sma(ta.stoch(rsi1, rsi1, rsi1, stochLen), smoothK)
stoch_d = ta.sma(stoch_k, smoothD)
[diPlus, diMinus, adxValue] = ta.dmi(adxLen, adxLen)
atrValue = ta.atr(atrLen)
htf_ema21 = request.security(syminfo.tickerid, "60", ta.ema(close, ema21Len), lookahead=barmerge.lookahead_off)
volAvg   = ta.sma(volume, 20)
volRatio = volAvg > 0 ? volume / volAvg : 0.0


// ========================== S/R DETECTION ===================================

pivHigh = ta.pivothigh(high, srLookback, srLookback)
pivLow  = ta.pivotlow(low, srLookback, srLookback)
var float[] srLevels = array.new_float(0)
if not na(pivHigh)
    array.push(srLevels, pivHigh)
    if array.size(srLevels) > 20
        array.shift(srLevels)
if not na(pivLow)
    array.push(srLevels, pivLow)
    if array.size(srLevels) > 20
        array.shift(srLevels)
nearSR   = float(na)
nearDist = 999999.0
if array.size(srLevels) > 0
    for i = 0 to array.size(srLevels) - 1
        lvl = array.get(srLevels, i)
        d   = math.abs(close - lvl)
        if d < nearDist
            nearDist := d
            nearSR   := lvl


// ========================== SESSION LOGIC ===================================

inSession    = not na(time(timeframe.period, "0930-1545", "America/New_York"))
isNewSession = inSession and not inSession[1]

// Minutes since 9:30 ET (0 at session open, grows through the day)
// Used for time-of-day analysis in the Data Lab
var int sessionMinute = 0
if isNewSession
    sessionMinute := 0
else if inSession
    sessionMinute := sessionMinute + 15  // 15-min bars


// ========================== SIGNAL CONDITIONS ===============================

bullStack = ema9 > ema21 and ema21 > ema50
bearStack = ema9 < ema21 and ema21 < ema50

vwapBull = close > vwapValue
vwapBear = close < vwapValue

htfBull = close > htf_ema21
htfBear = close < htf_ema21

// Squeeze: basic K vs D direction
stochBull = stoch_k > stoch_d
stochBear = stoch_k < stoch_d

// Trend: K > D AND K accelerating (v2.1 tightening)
trendStochBull = stoch_k > stoch_d and stoch_k > stoch_k[1]
trendStochBear = stoch_k < stoch_d and stoch_k < stoch_k[1]

adxOK = adxValue >= adxMin
volOK = volRatio >= volFloor

// Trend Continuation
trendLong  = vwapBull and bullStack and htfBull and trendStochBull and adxOK and volOK
trendShort = vwapBear and bearStack and htfBear and trendStochBear and adxOK and volOK

// Squeeze Detection
sqzDetected = false
for i = 0 to sqzLookback - 1
    if math.abs(ema9[i] - ema50[i]) < sqzThresh * atrValue[i]
        sqzDetected := true
        break

// Squeeze Entry (v2.1: dropped EMA50 gate)
sqzPriceLong  = close > ema9 and close > ema21
sqzPriceShort = close < ema9 and close < ema21
emaCrossUp    = ta.crossover(ema9, ema21)
emaCrossDown  = ta.crossunder(ema9, ema21)

sqzLong  = sqzDetected and sqzPriceLong  and emaCrossUp   and stochBull and volOK
sqzShort = sqzDetected and sqzPriceShort and emaCrossDown and stochBear and volOK


// ========================== REPUTATION ENGINE ===============================

ELIGIBLE = 0
GROUNDED = 1
EXTENDED = 2

type SignalTracker
    int    repState        = 0
    int    lockoutEnd      = 0
    int    consecStops     = 0
    bool   evalPending     = false
    float  evalPrice       = na
    float  evalATR         = na
    int    evalDir         = 0
    int    evalBar         = 0
    float  evalHigh        = na       // running high during eval window
    float  evalLow         = na       // running low during eval window
    bool   ghostPending    = false
    float  ghostPrice      = na
    float  ghostATR        = na
    int    ghostDir        = 0
    int    ghostBar        = 0
    float  ghostHigh       = na
    float  ghostLow        = na
    float  lastSignalPrice = na
    float  lastFollowTarget = na
    int    lastSignalDir   = 0
    int    entryBarCloseMs = 0       // write-once at ENTRY from time_close,
                                      // read at EVAL for parent_bar_close_ms

var SignalTracker trendTrk = SignalTracker.new()
var SignalTracker sqzTrk   = SignalTracker.new()
var bool sessionResetDone = false

if not inSession
    sessionResetDone := false
if isNewSession and not sessionResetDone
    trendTrk.repState       := ELIGIBLE
    trendTrk.lockoutEnd     := 0
    trendTrk.consecStops    := 0
    trendTrk.evalPending    := false
    trendTrk.ghostPending   := false
    trendTrk.lastSignalPrice  := na
    trendTrk.lastFollowTarget := na
    trendTrk.lastSignalDir  := 0
    sqzTrk.repState         := ELIGIBLE
    sqzTrk.lockoutEnd       := 0
    sqzTrk.consecStops      := 0
    sqzTrk.evalPending      := false
    sqzTrk.ghostPending     := false
    sqzTrk.lastSignalPrice   := na
    sqzTrk.lastFollowTarget  := na
    sqzTrk.lastSignalDir     := 0
    sessionResetDone         := true

// Update running high/low during active eval windows
if trendTrk.evalPending
    trendTrk.evalHigh := na(trendTrk.evalHigh) ? high : math.max(trendTrk.evalHigh, high)
    trendTrk.evalLow  := na(trendTrk.evalLow)  ? low  : math.min(trendTrk.evalLow,  low)
if trendTrk.ghostPending
    trendTrk.ghostHigh := na(trendTrk.ghostHigh) ? high : math.max(trendTrk.ghostHigh, high)
    trendTrk.ghostLow  := na(trendTrk.ghostLow)  ? low  : math.min(trendTrk.ghostLow,  low)
if sqzTrk.evalPending
    sqzTrk.evalHigh := na(sqzTrk.evalHigh) ? high : math.max(sqzTrk.evalHigh, high)
    sqzTrk.evalLow  := na(sqzTrk.evalLow)  ? low  : math.min(sqzTrk.evalLow,  low)
if sqzTrk.ghostPending
    sqzTrk.ghostHigh := na(sqzTrk.ghostHigh) ? high : math.max(sqzTrk.ghostHigh, high)
    sqzTrk.ghostLow  := na(sqzTrk.ghostLow)  ? low  : math.min(sqzTrk.ghostLow,  low)

checkFollowThru(int dir, float price, float atr) =>
    passed = false
    target = dir == 1 ? price + followThruMult * atr : price - followThruMult * atr
    for i = 0 to cooldownBars - 1
        if dir == 1 and high[i] >= target
            passed := true
        if dir == 2 and low[i] <= target
            passed := true
    passed

calcRR(float entry, float sl, float tp) =>
    risk = math.abs(entry - sl)
    reward = math.abs(tp - entry)
    risk > 0 ? math.round(reward / risk, 2) : 0.0


// ========================== WEBHOOK DATA VARIABLES ==========================
// These buffer the values that will be concatenated into the alert() JSON
// payload. They reset to 0 on each bar; only the bar that actually fires
// an ENTRY or EVAL writes meaningful values before the alert() call.

float sigDir        = 0.0
float sigType       = 0.0
float sigStatus     = 0.0
float sigSL         = 0.0
float sigTP1        = 0.0
float sigTP2        = 0.0
float sigStopPts    = 0.0
float sigRR1        = 0.0
float sigRR2        = 0.0
float sigStops      = 0.0
float sigRep        = 0.0
float sigFT         = 0.0
float sigResultCode = 0.0
float sigFTHigh     = 0.0
float sigFTLow      = 0.0
float sigFTActual   = 0.0
float sigMovePts    = 0.0
float sigStateBefore = 0.0
float sigStateAfter  = 0.0
float sigIsGhost    = 0.0
float sigHtfBull    = 0.0
float sigParentBarCloseMs = 0.0    // set at EVAL from tracker.entryBarCloseMs

bool fireTrendAlert = false
bool fireSqzAlert   = false
bool trendEntryLong  = false
bool trendEntryShort = false
bool sqzEntryLong    = false
bool sqzEntryShort   = false
bool trendEvalDone   = false
bool sqzEvalDone     = false


// ========================== PROCESS TREND TRACKER ===========================

// ---- 1. Real eval completion ----
if trendTrk.evalPending and (bar_index - trendTrk.evalBar >= cooldownBars)
    ftResult = checkFollowThru(trendTrk.evalDir, trendTrk.evalPrice, trendTrk.evalATR)
    stateBefore = trendTrk.repState
    if ftResult
        trendTrk.consecStops := 0
    else
        trendTrk.consecStops += 1
        if trendTrk.consecStops >= 2
            trendTrk.repState := EXTENDED
            trendTrk.lockoutEnd := bar_index + extendedBars
        else
            trendTrk.repState := GROUNDED
            trendTrk.lockoutEnd := bar_index + groundedBars

    ftTarget = trendTrk.evalDir == 1 ? trendTrk.evalPrice + followThruMult * trendTrk.evalATR : trendTrk.evalPrice - followThruMult * trendTrk.evalATR
    movePts  = trendTrk.evalDir == 1 ? close - trendTrk.evalPrice : trendTrk.evalPrice - close

    sigDir         := trendTrk.evalDir == 1 ? 1.0 : 2.0
    sigType        := 1.0
    sigStatus      := 2.0
    sigStops       := trendTrk.consecStops
    sigRep         := trendTrk.repState
    sigFT          := ftTarget
    sigResultCode  := ftResult ? 1.0 : 2.0
    sigFTHigh      := nz(trendTrk.evalHigh, high)
    sigFTLow       := nz(trendTrk.evalLow, low)
    sigFTActual    := close
    sigMovePts     := movePts
    sigStateBefore := stateBefore
    sigStateAfter  := trendTrk.repState
    sigIsGhost     := 0.0
    sigParentBarCloseMs := trendTrk.entryBarCloseMs
    fireTrendAlert := true
    trendEvalDone  := true
    trendTrk.evalPending := false
    trendTrk.evalHigh := na
    trendTrk.evalLow  := na

// ---- 2. Ghost eval completion ----
if trendTrk.ghostPending and (bar_index - trendTrk.ghostBar >= cooldownBars)
    ghostFT = checkFollowThru(trendTrk.ghostDir, trendTrk.ghostPrice, trendTrk.ghostATR)
    stateBeforeGhost = trendTrk.repState
    if ghostFT
        trendTrk.consecStops := math.max(0, trendTrk.consecStops - 1)
        if trendTrk.consecStops == 0
            trendTrk.repState := ELIGIBLE
            trendTrk.lockoutEnd := 0
        else
            trendTrk.repState := GROUNDED
            trendTrk.lockoutEnd := bar_index + groundedBars

    ghostTarget = trendTrk.ghostDir == 1 ? trendTrk.ghostPrice + followThruMult * trendTrk.ghostATR : trendTrk.ghostPrice - followThruMult * trendTrk.ghostATR
    ghostMove   = trendTrk.ghostDir == 1 ? close - trendTrk.ghostPrice : trendTrk.ghostPrice - close

    sigDir         := trendTrk.ghostDir == 1 ? 1.0 : 2.0
    sigType        := 1.0
    sigStatus      := 2.0
    sigStops       := trendTrk.consecStops
    sigRep         := trendTrk.repState
    sigFT          := ghostTarget
    sigResultCode  := ghostFT ? 1.0 : 2.0
    sigFTHigh      := nz(trendTrk.ghostHigh, high)
    sigFTLow       := nz(trendTrk.ghostLow, low)
    sigFTActual    := close
    sigMovePts     := ghostMove
    sigStateBefore := stateBeforeGhost
    sigStateAfter  := trendTrk.repState
    sigIsGhost     := 1.0
    sigParentBarCloseMs := trendTrk.entryBarCloseMs
    fireTrendAlert := true

    trendTrk.ghostPending := false
    trendTrk.ghostHigh := na
    trendTrk.ghostLow  := na

// ---- 3. Lockout expiry ----
if trendTrk.repState != ELIGIBLE and bar_index >= trendTrk.lockoutEnd
    trendTrk.repState := ELIGIBLE
    trendTrk.consecStops := 0

// ---- 4. Signal detection ----
trendHasLong  = trendLong  and inSession
trendHasShort = trendShort and inSession

if trendTrk.repState == ELIGIBLE and not trendTrk.evalPending and not fireTrendAlert
    if trendHasLong
        sl  = close - slMult  * atrValue
        tp1 = close + tp1Mult * atrValue
        tp2 = close + tp2Mult * atrValue
        sigDir     := 1.0
        sigType    := 1.0
        sigStatus  := 1.0
        sigSL      := sl
        sigTP1     := tp1
        sigTP2     := tp2
        sigStopPts := slMult * atrValue
        sigRR1     := calcRR(close, sl, tp1)
        sigRR2     := calcRR(close, sl, tp2)
        sigStops   := trendTrk.consecStops
        sigRep     := 0.0
        sigHtfBull := htfBull ? 1.0 : 0.0
        fireTrendAlert  := true
        trendEntryLong  := true

        trendTrk.evalPending     := true
        trendTrk.evalPrice       := close
        trendTrk.evalATR         := atrValue
        trendTrk.evalDir         := 1
        trendTrk.evalBar         := bar_index
        trendTrk.evalHigh        := high
        trendTrk.evalLow         := low
        trendTrk.lastSignalPrice  := close
        trendTrk.lastFollowTarget := close + followThruMult * atrValue
        trendTrk.lastSignalDir    := 1
        trendTrk.entryBarCloseMs  := time_close

    else if trendHasShort
        sl  = close + slMult  * atrValue
        tp1 = close - tp1Mult * atrValue
        tp2 = close - tp2Mult * atrValue
        sigDir     := 2.0
        sigType    := 1.0
        sigStatus  := 1.0
        sigSL      := sl
        sigTP1     := tp1
        sigTP2     := tp2
        sigStopPts := slMult * atrValue
        sigRR1     := calcRR(close, sl, tp1)
        sigRR2     := calcRR(close, sl, tp2)
        sigStops   := trendTrk.consecStops
        sigRep     := 0.0
        sigHtfBull := htfBull ? 1.0 : 0.0
        fireTrendAlert   := true
        trendEntryShort  := true

        trendTrk.evalPending     := true
        trendTrk.evalPrice       := close
        trendTrk.evalATR         := atrValue
        trendTrk.evalDir         := 2
        trendTrk.evalBar         := bar_index
        trendTrk.evalHigh        := high
        trendTrk.evalLow         := low
        trendTrk.lastSignalPrice  := close
        trendTrk.lastFollowTarget := close - followThruMult * atrValue
        trendTrk.lastSignalDir    := 2
        trendTrk.entryBarCloseMs  := time_close

// ---- 5. Ghost signal detection ----
if trendTrk.repState != ELIGIBLE and not trendTrk.ghostPending
    if trendHasLong
        trendTrk.ghostPending     := true
        trendTrk.ghostPrice       := close
        trendTrk.ghostATR         := atrValue
        trendTrk.ghostDir         := 1
        trendTrk.ghostBar         := bar_index
        trendTrk.ghostHigh        := high
        trendTrk.ghostLow         := low
        trendTrk.lastSignalPrice  := close
        trendTrk.lastFollowTarget := close + followThruMult * atrValue
        trendTrk.lastSignalDir    := 1
        trendTrk.entryBarCloseMs  := time_close
    else if trendHasShort
        trendTrk.ghostPending     := true
        trendTrk.ghostPrice       := close
        trendTrk.ghostATR         := atrValue
        trendTrk.ghostDir         := 2
        trendTrk.ghostBar         := bar_index
        trendTrk.ghostHigh        := high
        trendTrk.ghostLow         := low
        trendTrk.lastSignalPrice  := close
        trendTrk.lastFollowTarget := close - followThruMult * atrValue
        trendTrk.lastSignalDir    := 2
        trendTrk.entryBarCloseMs  := time_close


// ========================== PROCESS SQUEEZE TRACKER =========================

// ---- 1. Real eval completion ----
if sqzTrk.evalPending and (bar_index - sqzTrk.evalBar >= cooldownBars)
    ftResult = checkFollowThru(sqzTrk.evalDir, sqzTrk.evalPrice, sqzTrk.evalATR)
    stateBefore = sqzTrk.repState
    if ftResult
        sqzTrk.consecStops := 0
    else
        sqzTrk.consecStops += 1
        if sqzTrk.consecStops >= 2
            sqzTrk.repState := EXTENDED
            sqzTrk.lockoutEnd := bar_index + extendedBars
        else
            sqzTrk.repState := GROUNDED
            sqzTrk.lockoutEnd := bar_index + groundedBars

    ftTarget = sqzTrk.evalDir == 1 ? sqzTrk.evalPrice + followThruMult * sqzTrk.evalATR : sqzTrk.evalPrice - followThruMult * sqzTrk.evalATR
    movePts  = sqzTrk.evalDir == 1 ? close - sqzTrk.evalPrice : sqzTrk.evalPrice - close

    sigDir         := sqzTrk.evalDir == 1 ? 1.0 : 2.0
    sigType        := 2.0
    sigStatus      := 2.0
    sigStops       := sqzTrk.consecStops
    sigRep         := sqzTrk.repState
    sigFT          := ftTarget
    sigResultCode  := ftResult ? 1.0 : 2.0
    sigFTHigh      := nz(sqzTrk.evalHigh, high)
    sigFTLow       := nz(sqzTrk.evalLow, low)
    sigFTActual    := close
    sigMovePts     := movePts
    sigStateBefore := stateBefore
    sigStateAfter  := sqzTrk.repState
    sigIsGhost     := 0.0
    sigParentBarCloseMs := sqzTrk.entryBarCloseMs
    fireSqzAlert := true
    sqzEvalDone  := true
    sqzTrk.evalPending := false
    sqzTrk.evalHigh := na
    sqzTrk.evalLow  := na

// ---- 2. Ghost eval completion ----
if sqzTrk.ghostPending and (bar_index - sqzTrk.ghostBar >= cooldownBars)
    ghostFT = checkFollowThru(sqzTrk.ghostDir, sqzTrk.ghostPrice, sqzTrk.ghostATR)
    stateBeforeGhost = sqzTrk.repState
    if ghostFT
        sqzTrk.consecStops := math.max(0, sqzTrk.consecStops - 1)
        if sqzTrk.consecStops == 0
            sqzTrk.repState := ELIGIBLE
            sqzTrk.lockoutEnd := 0
        else
            sqzTrk.repState := GROUNDED
            sqzTrk.lockoutEnd := bar_index + groundedBars

    ghostTarget = sqzTrk.ghostDir == 1 ? sqzTrk.ghostPrice + followThruMult * sqzTrk.ghostATR : sqzTrk.ghostPrice - followThruMult * sqzTrk.ghostATR
    ghostMove   = sqzTrk.ghostDir == 1 ? close - sqzTrk.ghostPrice : sqzTrk.ghostPrice - close

    sigDir         := sqzTrk.ghostDir == 1 ? 1.0 : 2.0
    sigType        := 2.0
    sigStatus      := 2.0
    sigStops       := sqzTrk.consecStops
    sigRep         := sqzTrk.repState
    sigFT          := ghostTarget
    sigResultCode  := ghostFT ? 1.0 : 2.0
    sigFTHigh      := nz(sqzTrk.ghostHigh, high)
    sigFTLow       := nz(sqzTrk.ghostLow, low)
    sigFTActual    := close
    sigMovePts     := ghostMove
    sigStateBefore := stateBeforeGhost
    sigStateAfter  := sqzTrk.repState
    sigIsGhost     := 1.0
    sigParentBarCloseMs := sqzTrk.entryBarCloseMs
    fireSqzAlert := true

    sqzTrk.ghostPending := false
    sqzTrk.ghostHigh := na
    sqzTrk.ghostLow  := na

// ---- 3. Lockout expiry ----
if sqzTrk.repState != ELIGIBLE and bar_index >= sqzTrk.lockoutEnd
    sqzTrk.repState := ELIGIBLE
    sqzTrk.consecStops := 0

// ---- 4. Signal detection ----
sqzHasLong  = sqzLong  and inSession
sqzHasShort = sqzShort and inSession

if sqzTrk.repState == ELIGIBLE and not sqzTrk.evalPending and not fireSqzAlert
    if sqzHasLong
        sl  = close - slMult  * atrValue
        tp1 = close + tp1Mult * atrValue
        tp2 = close + tp2Mult * atrValue
        sigDir     := 1.0
        sigType    := 2.0
        sigStatus  := 1.0
        sigSL      := sl
        sigTP1     := tp1
        sigTP2     := tp2
        sigStopPts := slMult * atrValue
        sigRR1     := calcRR(close, sl, tp1)
        sigRR2     := calcRR(close, sl, tp2)
        sigStops   := sqzTrk.consecStops
        sigRep     := 0.0
        sigHtfBull := htfBull ? 1.0 : 0.0
        fireSqzAlert := true
        sqzEntryLong := true

        sqzTrk.evalPending     := true
        sqzTrk.evalPrice       := close
        sqzTrk.evalATR         := atrValue
        sqzTrk.evalDir         := 1
        sqzTrk.evalBar         := bar_index
        sqzTrk.evalHigh        := high
        sqzTrk.evalLow         := low
        sqzTrk.lastSignalPrice  := close
        sqzTrk.lastFollowTarget := close + followThruMult * atrValue
        sqzTrk.lastSignalDir    := 1
        sqzTrk.entryBarCloseMs  := time_close

    else if sqzHasShort
        sl  = close + slMult  * atrValue
        tp1 = close - tp1Mult * atrValue
        tp2 = close - tp2Mult * atrValue
        sigDir     := 2.0
        sigType    := 2.0
        sigStatus  := 1.0
        sigSL      := sl
        sigTP1     := tp1
        sigTP2     := tp2
        sigStopPts := slMult * atrValue
        sigRR1     := calcRR(close, sl, tp1)
        sigRR2     := calcRR(close, sl, tp2)
        sigStops   := sqzTrk.consecStops
        sigRep     := 0.0
        sigHtfBull := htfBull ? 1.0 : 0.0
        fireSqzAlert  := true
        sqzEntryShort := true

        sqzTrk.evalPending     := true
        sqzTrk.evalPrice       := close
        sqzTrk.evalATR         := atrValue
        sqzTrk.evalDir         := 2
        sqzTrk.evalBar         := bar_index
        sqzTrk.evalHigh        := high
        sqzTrk.evalLow         := low
        sqzTrk.lastSignalPrice  := close
        sqzTrk.lastFollowTarget := close - followThruMult * atrValue
        sqzTrk.lastSignalDir    := 2
        sqzTrk.entryBarCloseMs  := time_close

// ---- 5. Ghost signal detection ----
if sqzTrk.repState != ELIGIBLE and not sqzTrk.ghostPending
    if sqzHasLong
        sqzTrk.ghostPending     := true
        sqzTrk.ghostPrice       := close

// ============================================================================
// HEARTBEAT SLOT MACHINERY — Option C §7
// ============================================================================
// Additive only. Does not modify the Reputation Engine, signal logic, filters,
// session window, or the alert firing conditions. Slot state is measurement
// infrastructure (per §1.4 discipline rule) — Pine records positions the
// Reputation Engine already authorized, but does not re-decide.
//
// Four slots max (§3.4 / OI-04: N=4, revisit on observation data). Slot
// release uses the local closure heuristic (§7.5): original SL breach,
// TP2 breach, or session end. BE_STOP and EOD_TIMEOUT closures are handled
// by the backend resolver — Pine's wasted heartbeats are a safe no-op at
// the backend per §5.12 TC12.

var int   posSlot1Time = 0
var float posSlot1Sl   = 0.0
var float posSlot1Tp2  = 0.0
var int   posSlot1Dir  = 0

var int   posSlot2Time = 0
var float posSlot2Sl   = 0.0
var float posSlot2Tp2  = 0.0
var int   posSlot2Dir  = 0

var int   posSlot3Time = 0
var float posSlot3Sl   = 0.0
var float posSlot3Tp2  = 0.0
var int   posSlot3Dir  = 0

var int   posSlot4Time = 0
var float posSlot4Sl   = 0.0
var float posSlot4Tp2  = 0.0
var int   posSlot4Dir  = 0

// ---- §7.4 Slot addition on ENTRY ----
// Runs AFTER the Reputation Engine has set fireTrendAlert / fireSqzAlert and
// populated sigDir / sigSL / sigTP2. Lowest-indexed empty slot wins.
if (fireTrendAlert or fireSqzAlert) and sigStatus == 1.0
    nowMs  = time_close
    nowSl  = sigSL
    nowTp2 = sigTP2
    nowDir = int(sigDir)
    if posSlot1Time == 0
        posSlot1Time := nowMs
        posSlot1Sl   := nowSl
        posSlot1Tp2  := nowTp2
        posSlot1Dir  := nowDir
    else if posSlot2Time == 0
        posSlot2Time := nowMs
        posSlot2Sl   := nowSl
        posSlot2Tp2  := nowTp2
        posSlot2Dir  := nowDir
    else if posSlot3Time == 0
        posSlot3Time := nowMs
        posSlot3Sl   := nowSl
        posSlot3Tp2  := nowTp2
        posSlot3Dir  := nowDir
    else if posSlot4Time == 0
        posSlot4Time := nowMs
        posSlot4Sl   := nowSl
        posSlot4Tp2  := nowTp2
        posSlot4Dir  := nowDir
    // else: overflow (§7.5.2) — backend staleness check will catch it.

// ---- §7.5 Slot release — local closure heuristic ----
// A slot is released when the bar's range crossed original SL or TP2.
// BE stops post-TP1 and EOD timeouts are NOT detected here by design.
releaseSlot1 = posSlot1Time != 0 and ((posSlot1Dir == 1 and (low <= posSlot1Sl or high >= posSlot1Tp2)) or (posSlot1Dir == 2 and (high >= posSlot1Sl or low <= posSlot1Tp2)))
releaseSlot2 = posSlot2Time != 0 and ((posSlot2Dir == 1 and (low <= posSlot2Sl or high >= posSlot2Tp2)) or (posSlot2Dir == 2 and (high >= posSlot2Sl or low <= posSlot2Tp2)))
releaseSlot3 = posSlot3Time != 0 and ((posSlot3Dir == 1 and (low <= posSlot3Sl or high >= posSlot3Tp2)) or (posSlot3Dir == 2 and (high >= posSlot3Sl or low <= posSlot3Tp2)))
releaseSlot4 = posSlot4Time != 0 and ((posSlot4Dir == 1 and (low <= posSlot4Sl or high >= posSlot4Tp2)) or (posSlot4Dir == 2 and (high >= posSlot4Sl or low <= posSlot4Tp2)))
if releaseSlot1
    posSlot1Time := 0
    posSlot1Sl   := 0.0
    posSlot1Tp2  := 0.0
    posSlot1Dir  := 0
if releaseSlot2
    posSlot2Time := 0
    posSlot2Sl   := 0.0
    posSlot2Tp2  := 0.0
    posSlot2Dir  := 0
if releaseSlot3
    posSlot3Time := 0
    posSlot3Sl   := 0.0
    posSlot3Tp2  := 0.0
    posSlot3Dir  := 0
if releaseSlot4
    posSlot4Time := 0
    posSlot4Sl   := 0.0
    posSlot4Tp2  := 0.0
    posSlot4Dir  := 0

// ---- §7.5.3 Session end clears all slots ----
sessionEnded = inSession[1] and not inSession
if sessionEnded
    posSlot1Time := 0
    posSlot1Sl   := 0.0
    posSlot1Tp2  := 0.0
    posSlot1Dir  := 0
    posSlot2Time := 0
    posSlot2Sl   := 0.0
    posSlot2Tp2  := 0.0
    posSlot2Dir  := 0
    posSlot3Time := 0
    posSlot3Sl   := 0.0
    posSlot3Tp2  := 0.0
    posSlot3Dir  := 0
    posSlot4Time := 0
    posSlot4Sl   := 0.0
    posSlot4Tp2  := 0.0
    posSlot4Dir  := 0
